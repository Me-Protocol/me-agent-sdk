Quick Start

Add the SDK to your HTML page:

<!-- Load the SDK -->
<script src="https://cdn.jsdelivr.net/gh/YOUR_USERNAME/me-agent-sdk@main/dist/me-agent-sdk.min.js"></script>

<!-- Initialize -->
<script>
  MeAgent.init({
    emailAddress: 'user@example.com',
    brandId: 'your-brand-id',
    userId: 'your-user-id',
    environment: MeAgent.Environment.DEV,
    network: MeAgent.Network.SEPOLIA
  });
</script>
